Good JUnit BDD-style test

public class MapTest {

    @Test
    public void shouldAddAllItemsFromOneListToAnother() {
        //given:
        List<String> baseList = Lists.newArrayList("one", "two");
        List<String> addMe = Lists.newArrayList("three", "four");

        //when:
        baseList.addAll(addMe);

        //then:
        Assert.assertEquals(Arrays.asList("one", "two", "three", "four"),
            baseList);
    }
}

Data Driven Tests

Data driven

@Unroll
def """should round #givenValue.class.simpleName #givenValue
       to #expectedValue with scale #scale"""() {

    when:
        def result = Money.round(givenValue, scale)

    then:
        result == expectedValue

    where:
        givenValue                       | scale || expectedValue
        0.339d                           | 2     || 0.34
        0.22d                            | 2     || 0.22
        0.1d                             | 1     || 0.1
        new BigDecimal(2.333, DECIMAL32) | 2     || 2.33
}

Mock vs Stub

  • Mock - is for verifying interactions
  • Stub - provides canned responses
    • we teach Stubs!

Mock

def "should check that money fall into coin cassette"() {

    given:
        def coinCassetteMock = Mock(CoinCassette)
        def vendingMachine = new VendingMachine(Stub(ProductMagazine),
                                                coinCassetteMock)

    when:
        vendingMachine.insert QUARTER

    then:
        1 * coinCassetteMock.push(QUARTER)
}

Stub

def "should return coin when coin cassette is full"() {

    given:
        def coinCassetteMock = Stub(CoinCassette) {
            isFull(_ as Coin) >> true
        }

        def vendingMachine = new VendingMachine(Stub(ProductMagazine),
                                                coinCassetteMock)

    when:
        def returnedCoin = vendingMachine.insert QUARTER

    then:
        returnedCoin.get() == QUARTER
}

def "should show groovy goodnes"() {

    given:
        def jsonBuilder = new JsonBuilder()
        jsonBuilder {
            integerList 1, 2, 3, 4, 5
            stringsMap  {
                id1  "Spock"
                id2  "TestNG"
                id3  "JUnit"
            }
        }

    when:
        def json = new JsonSlurper().parseText jsonBuilder.toString()

    then:
        json.integerList == [1, 2, 3, 4, 5]
        json.stringsMap == [id1: "Spock", id2: "TestNG", id3: "JUnit"]
}
