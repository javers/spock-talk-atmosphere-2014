Good JUnit BDD-style test

public class MapTest {

    @Test
    public void shouldStoreValueInHashMapAndRetrieveItByKey() {
        //given:
        Map<String,Integer> map = new HashMap<>();

        //when:
        map.put("myKey", 5);

        //then:
        Assert.assertEquals(map.get("myKey"), 5);
        Assert.assertNull(map.get("anything"));
    }
}

Spock BDD Test

class MapTest extends Specification {

    def "should store value in HashMap and retrieve it by key" () {
        given:
            def map = new HashMap()

        when:
            map.put("myKey", 5)

        then:
            map.myKey == 5
            map.anything == null
    }
}

Assertions in then/expect section


    def "should evaluate assertions with Groovy Truth"() {

    expect:
    //boolean expresions
    assert true
    !false

    // Collection & maps
    [1]          //non-empty collection -> true
    ['one': 1]   //non-empty map        -> true

    ![]          //empty collection     -> false
    ![:]         //empty map            -> false

    // Number
    1            //non-zero             -> true
    !0           //zero                 -> false

    // Strings
    'a'          //non-empty            -> true
    ' '          //non-empty            -> true
    !''          //empty                -> false

    // Object references
    this         //not null             -> true
    !null        //null                 -> false
}

@Unroll
def "should round #givenValue.class.simpleName #givenValue to #expectedValue"() {

    when:
    def result = Money.round2(givenValue)

    then:
    result == expectedValue

    where:
    givenValue                       | expectedValue
    0.339d                           | 0.34
    0.22d                            | 0.22
    0.1d                             | 0.1
    new BigDecimal(2.333, DECIMAL32) | 2.33
}
